Definition of anatomical areas

Definition of anatomical areas

round_base <- function(x, base = 50) {
  return(round(x / base) * base)
}

# First we make the squares of 50*50 microns.
df_data <- df_data %>% 
  mutate(X2 = round_base(X, 50), Y2 = round_base(Y, 50))

# Here we count the tumor cells within each square:
tumor_fraction_superpixels <- df_data %>% 
  group_by(slide_id, scene_id, X2, Y2) %>%
  summarise(N = sum(CellType == "Tumor"), M = n()) %>%
  ungroup() %>%
  mutate(P = 100*N/M)

# Here we divide the cores into four regions based on the cell counts:
tumor_fraction_superpixels <- tumor_fraction_superpixels %>% 
  mutate(region = ifelse(N > 12, 'high_density', 
                         ifelse(N > 8, 'medium density', 
                                ifelse(N >= 1, 'low density', 'non-tumor'))))

df_data <- df_data %>% left_join(tumor_fraction_superpixels %>% select(slide_id, scene_id, X2,Y2, region)) %>% rename(tumor_region = region)

# Here we count the blank cells within each square for the low density and non tumoral areas:
blank_fraction_superpixels <- df_data %>% 
  filter(tumor_region %in% c('low density', 'non-tumor')) %>% 
  group_by(slide_id, scene_id, X2, Y2) %>%
  summarise(N = sum(CellType == "NOS"), M = n()) %>%
  ungroup() %>%
  mutate(P = 100*N/M)

# Here we divide the cores into four regions based on the cell counts:
blank_fraction_superpixels <- blank_fraction_superpixels %>% 
  mutate(region = ifelse(N > 9, 'high_density', 
                         ifelse(N > 5, 'medium density', 
                                ifelse(N > 1, 'low density', 'non-blank'))))

df_data <- df_data %>% left_join(blank_fraction_superpixels %>% rename(blank_region = region) %>% select(slide_id, scene_id, X2, Y2,blank_region)) 

# Here we merge both classifications into four final categories:
df_data <- df_data  %>%
  mutate(region = tumor_region) %>%  
  mutate(region = ifelse(blank_region %in% 'high_density', 'Necrotic', 
                         ifelse(region %in% c('high_density','medium density'),'High_tumor', region)))%>% 
  mutate(region = ifelse(region %in% "High_tumor","High Tumoral Density", 
                         ifelse(region %in% "low density", "Low Tumoral Density", 
                                ifelse(region %in% "non-tumor", "Non-Tumoral", "Necrotic Area"))))

Comparison of anatomical regions between survival groups

Landscape and comparison between survival groups

df_landscape <- df_areas %>% 
  group_by(survival_group, couple_id, patient_id, region) %>% 
  summarise(N = n()) %>% 
  ungroup() %>% 
  spread(region, N, fill = 0) %>% 
  gather(region, N, -survival_group, -couple_id, -patient_id) %>% 
  group_by(survival_group, patient_id) %>% 
  mutate(M = sum(N)) %>% 
  ungroup() %>% 
  mutate(P = 100*N/M)

ggplot(df_landscape, aes(paste(survival_group, patient_id), P, fill = region)) + geom_col() + theme_bw() + facet_grid(~couple_id, scales = 'free_x', space = 'free_x') + theme_bw() + theme(legend.position = 'bottom') + theme(axis.text.x = element_text(angle = 90, vjust = 0.5, hjust=1)) + ylab('Region Percentage (%)') + xlab('patient identifier')

Treated vs untreated samples

Landscape and comparison between survival groups per treatment category (untreated - 0 vs treated - 1).

df_landscape <- df_areas %>% 
  group_by(survival_group, couple_id, patient_id, treated, region) %>% 
  summarise(N = n()) %>% 
  ungroup() %>% 
  spread(region, N, fill = 0) %>% 
  gather(region, N, -survival_group, -couple_id, -patient_id, -treated) %>% 
  group_by(survival_group, patient_id, treated) %>% 
  mutate(M = sum(N)) %>% 
  ungroup() %>% 
  mutate(P = 100*N/M)

ggplot(df_landscape, aes(paste(survival_group, patient_id), P, fill = region)) + geom_col() + theme_bw() + facet_grid(~couple_id+treated, scales = 'free_x', space = 'free_x') + theme_bw() + theme(legend.position = 'bottom') + theme(axis.text.x = element_text(angle = 90, vjust = 0.5, hjust=1)) + ylab('Region Percentage (%)') + xlab('patient identifier')

library(ggpubr)
library(rstatix)

stat.test <- df_landscape %>%
  filter(M > 0) %>%
  group_by(region, treated) %>%
  mutate(QC = sum(N>0)) %>%
  ungroup() %>%
  filter(QC>0) %>%
  select(-QC) %>%
  group_by(region, treated) %>%
  wilcox_test(P ~ survival_group) %>%
  adjust_pvalue(method = "fdr") %>%
  add_significance() %>% 
  left_join(df_landscape %>% group_by(region, treated) %>% summarise(Pmax = 0.9*max(P)) %>% ungroup()) %>% 
  mutate(p.adj = round(p.adj, 4))

DT::datatable(stat.test, filter = 'top')
ggboxplot(df_landscape, x = "survival_group", y = "P", fill = "survival_group") + #, palette = "jco", # ylim = c(0, 40)
  theme_bw() + 
  geom_jitter(alpha = 0.2) +
  scale_fill_manual(values = c('royalblue', 'goldenrod')) +
  theme(legend.position = 'none') +
  xlab(NULL) + ylab('Relative Area Fraction (%)') +
  facet_wrap(~region+treated, scales = 'free_y', nrow = 1) +
  stat_pvalue_manual(stat.test, label = "p", y.position = 'Pmax')
